% http://tex.stackexchange.com/questions/38208/beamer-create-own-headline-theme
%
% !!! Commented linebreaks are necessary !!!

\setbeamertemplate{headline}
{%
    \leavevmode%
    \hbox{%
        \begin{beamercolorbox}[wd=\paperwidth,ht=5.7ex,dp=2.2ex]{headline}%
            \raggedright\hspace*{2em}%
            {%
                \ifx\insertsubsection\empty
                    \raise1pt\hbox{\normalsize\thesection.\enspace\insertsection}%
                \else
                    \raise1pt\hbox{\normalsize\thesection.{\footnotesize\thesubsection}.\enspace\normalsize\insertsubsection}%
                \fi
                \hfill%
                {%
                    \normalsize%
                    \raise1pt\hbox{%
                        \raise2pt\hbox{\insertframenumber}\color{gray}{/}%
                        \raise-2pt\hbox{\inserttotalframenumber}
                    }
                }%
            }%
            \hspace*{2em}%
        \end{beamercolorbox}%
    }%
}

% customized title page layout
\defbeamertemplate*{title page}{customized}[1][]
{
    \centering
    \par
    \usebeamercolor[fg]{normal text}
    \usebeamerfont{institute}\insertinstitute\par
    %\resizebox{!}{2cm}{\inserttitlegraphic}
    \vfill
    \begin{beamercolorbox}[wd=\paperwidth]{headline}
    {
        \par\smallskip
        \usebeamerfont{title}{\hfill\inserttitle\hfill}
        \par\bigskip
    }
    \end{beamercolorbox}
    \vskip-2pt
    \begin{beamercolorbox}[wd=\paperwidth]{frametitle}
        \par\smallskip
        \usebeamerfont{subtitle}{\hfill\insertsubtitle\hfill}
        \par\medskip
    \end{beamercolorbox}
    \vfill
    {
        \usebeamercolor[fg]{normal text}
        \small
        \usebeamerfont{author}\textbf{\insertauthor{}}
        \par
        \href{mailto:\presentationAuthorEmail}{\presentationAuthorEmail}
        \par
        \adforn{45}
        \par\medskip
        \usebeamerfont{date}\insertdate
    }
    \par
}
